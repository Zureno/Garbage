        function resetTimeout() {
            $("#timeoutMaster").hide();
            $.post(rootUrl + "session/keepalive", function (info) {
                if (info.active == false || info.expiresIn <= 0) {
                    logOut();
                    return;
                }
                scheduleTimeout(info.expiresIn);
            });
        }

        function checkTimeout() {
            $.post(rootUrl + "session/info", { id: $("#timeoutMaster").attr("data-id") }, function (info) {
                if (info.active == false || info.expiresIn <= 0) {
                    logOut();
                    return;
                }
                scheduleTimeout(info.expiresIn);
            });
        }

        var timeoutTimer = 0;
        function scheduleTimeout(expiresIn) {
            clearTimeout(timeoutTimer);
            if (expiresIn <= 120) {
                $("#timeoutMaster").show();
                timeoutTimer = setTimeout(checkTimeout, expiresIn * 1000);
            } else {
                $("#timeoutMaster").hide();
                timeoutTimer = setTimeout(checkTimeout, (expiresIn - 120) * 1000);
            }
        }
