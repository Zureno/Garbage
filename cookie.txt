$(document).ready(function () {

    isOnfocus = window.document.hasFocus();

    $(".get-id-txt").focus(function () {
        $(".get-id-drop-down").addClass("menu-hover");
        $(".get-id-drop-down .nav-dropdown-content").addClass("display-block");
    });
    $("#getClientIdSelect").focus(function () {
        $(".get-id-drop-down").addClass("menu-hover");
        $(".get-id-drop-down .nav-dropdown-content").addClass("display-block");
    });
    $(document).on("click", function (event) {
        // Check if the clicked element is inside the dropdown
        if ($(event.target).closest(".nav-dropdown.get-id-drop-down").length) {
            $(".get-id-drop-down").addClass("menu-hover");
            $(".get-id-drop-down .nav-dropdown-content").addClass("display-block");
        } else {
            $(".get-id-drop-down").removeClass("menu-hover");
            $(".get-id-drop-down .nav-dropdown-content").removeClass("display-block");
        }
    });
    $("#getIdTextbox").keyup(enterKeyPressed);
    $(".client-dropdown > li").click(clientPressed);
    $("#clientSelectSearch").keyup(enterPressedOnClientSearch);
    keyPressOnClientSelectEvent();

    new Fingerprint2().get(function (result, components) {
        var canvasFingerprinting = "";
        if (typeof window.console !== "undefined") {
            for (var index in components) {
                var obj = components[index];
                if (obj.key === "canvas") {
                    canvasFingerprinting = obj.value.toString().substr(0, 1000);
                    break;
                }
            }
            //setFingerprinting(canvasFingerprinting);
            $.ajax({
                type: "POST",
                url: rootUrl + "WebService/SetFingerprinting",
                data: { fingerprinting: canvasFingerprinting },
                success: function (responseHtml) {
                },
                error: function (data) {
                }
            });
        }
    });
});

function setTabDisplay(divTab, divTabMiddle, selected) {
    var outerDiv = $('#' + divTab);
    var innerDiv = $('#' + divTabMiddle);
    if (outerDiv && innerDiv) {
        if (selected === true) {
            $(outerDiv).css("display", "block");
            $(innerDiv).attr("class", "tabTitle selectedTab");
        }
        else {
            $(outerDiv).css("display", "none");
            $(innerDiv).attr("class", "tabTitle unSelectedTab");
        }
    }
}

function downloadDocument(fileId) {
    frm = document.forms[0];
    txtDoc = document.getElementById('<% =txtDocumentFileId.ClientID %>');
    txtDoc.value = fileId;
    frm.target = "_blank";
    frm.submit();
    txtDoc.value = "";
    frm.target = "";
}

function DisableEnterKey(event) {
    key = event.keyCode ? event.keyCode : event.which ? event.which : event.charCode;
    if (key === 13 && (!event.target || event.target.tagName !== "TEXTAREA") && allowEnter === false) {
        return false;
    }
    else {
        return true;
    }
}

function openPage(pageToOpen, clickedLink, addArg) {
    document.frmNAV.txtTreeViewSelection.value = clickedLink;
    document.frmNAV.action = pageToOpen;
    document.frmNAV.txtAdditionalArg.value = addArg;
    document.frmNAV.submit();
}

function showChildMenu(id, event) {
    event.stopPropagation();
    var childMenu = document.getElementById(id);
    if (childMenu.style.display === "none") {
        childMenu.style.display = "block";
        $(event.target).attr("class", "menuDownArrow");
        addExpandedMenu(id);
    }
    else {
        childMenu.style.display = "none";
        $(event.target).attr("class", "menuRightArrow");
        removeExpandedMenu(id);
    }
}

var expandedMenus = Array();

function fillExpandedMenusArray() {
    var menus = document.forms[0].ctl00_txtExpandedMenus.value.split("|");
    expandedMenus = menus;
    document.frmNAV.txtExpandedMenus.value = document.forms[0].ctl00_txtExpandedMenus.value;
}

function addExpandedMenu(menu) {
    var expandedCount = expandedMenus.length;
    expandedMenus[expandedCount] = menu;
    saveExpandedMenus();
}

function removeExpandedMenu(menu) {
    var expandedCount = expandedMenus.length;
    for (var i = 0; i < expandedCount; i++) {
        if (menu === expandedMenus[i].toString()) {
            expandedMenus.splice(i, 1);
            break;
        }
    }
    saveExpandedMenus();
}

function saveExpandedMenus() {
    var expandedCount = expandedMenus.length;
    var menus = "";
    for (var i = 0; i < expandedCount; i++) {
        menus += expandedMenus[i].toString() + "|";
    }
    document.frmNAV.txtExpandedMenus.value = menus;
}

function enterKey() {
    if (allowEnter === false) {
        allowEnter = true;
    }
    else {
        allowEnter = false;
    }
}

//Fix Firefox extra line in textareas
$("textarea").each(function () {
    var lineHeight = parseFloat($(this).css("line-height"));
    var lines = $(this).attr("rows") * 1 || $(this).prop("rows") * 1;
    $(this).css("height", lines * lineHeight);
});

function enterKeyPressed(e) {
    if (e.keyCode === 13) {
        if (e.target.id === "getIdTextbox" && e.target.value.length !== 0) {
            searchGETID(e.target.value, getClientIdDropdownValue());
        }
    }
}

function getIdTextboxValue() {
    return document.getElementById("getIdTextbox").value;
}

function getClientIdDropdownValue() {
    return document.getElementById("getClientIdSelect").value;
}

function searchGETID(searchText, clientId) {
    console.log("searchGETID:" + searchText);
    document.quickSearchForm.action = rootUrl + "Home/QuickSearch";
    document.quickSearchForm.method = "post";
    document.quickSearchForm.quickSearchType.value = "GETID";
    document.quickSearchForm.quickSearchValue.value = searchText;
    document.quickSearchForm.quickSearchClientId.value = clientId;
    document.quickSearchForm.submit();
}

function enableClientSearchBox() {
    $("#clientSelectSearch").width($("#lblCurrentClient").width());
    $("#lblCurrentClient").hide();
    $("#clientSelectSearch").show();
    $("#clientSelectSearch").focus();
}

function showlblCurrentClient() {
    //if there is no hover, this means that the mouse is not over the client dropdown anymore
    if (!($(".client-dropdown > li:hover").length != 0)) {
        $("#clientSelectSearch").hide();
        $("#clientSelectSearch").val("");
        $("#lblCurrentClient").show();
        $(".client-dropdown > li").show();
    }
}

function filterClientSelect(element) {
    var value = $(element).val().toUpperCase();

    $(".client-dropdown > li").each(function () {
        if ($(this).find("a").text().toUpperCase().search(value) > -1) {
            $(this).show();
        }
        else {
            $(this).hide();
        }
    });
}

function switchClient(clientId, clientCode) {
    var elmnt = document.getElementsByClassName('js-switch-client-id');
    var elmntCode = document.getElementsByClassName('js-switch-client-code');
    if (elmnt != null) {
        elmnt[0].value = clientId;
        elmntCode[0].value = clientCode;
        elmnt[0].form.submit();
    }
}

function clientPressed(event) {

    var clientId = $(event.target).attr("data-client-id");
    var clientCode = $(event.target).attr("data-client-code");
    if (event.target.nodeName == "A")//added this because sometime if the anchor tag is pressed it won't be able to get the data-client-id which is on the li tag
    {
        clientId = $(event.target).parent().attr("data-client-id");
        clientCode = $(event.target).parent().attr("data-client-code");
    }
    if (clientId) {
        switchClient(clientId, clientCode);
    }
}

function enterPressedOnClientSearch(event) {
    var elementId = event.target.id;
    if (event.keyCode == 13) {
        if (elementId == "clientSelectSearch") {
            var clientId = $(".client-dropdown > li:visible").first().attr("data-client-id");
            var clientCode = $(".client-dropdown > li:visible").first().attr("data-client-code");
            if (clientId) {
                switchClient(clientId, clientCode);
            }
        }
    }
}

function keyPressOnClientSelectEvent() {
    var lblCurrentClient = document.getElementById('lblCurrentClient');
    var ulClientSelect = document.getElementsByClassName('client-dropdown')[0];

    function keyaction(e) {
        enableClientSearchBox();
    }
    if (lblCurrentClient) {
        lblCurrentClient.addEventListener("mouseover", function () {
            document.addEventListener("keypress", keyaction);
        });

        lblCurrentClient.addEventListener("mouseout", function () {
            document.removeEventListener("keypress", keyaction, false);
        });
    }

    if (ulClientSelect) {
        ulClientSelect.addEventListener("mouseover", function () {
            document.addEventListener("keypress", keyaction);
        });

        ulClientSelect.addEventListener("mouseout", function () {
            document.removeEventListener("keypress", keyaction, false);
        });
    }
}
