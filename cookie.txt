        var searchQuery = message.toString().replace("searchGETID:", "").trim();
        capturedSearches.push({
            timestamp: new Date(),
            searchQuery: searchQuery,
            url: window.location.href
        });
        console.warn(" CAPTURED SEARCH:", searchQuery);
    }
    originalLog.apply(console, arguments);
};

            searchText: searchText,
            clientId: clientId,
