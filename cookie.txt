akamaivalidator Lambda Handler - Enhanced Edition

Uses UCAS Debugger Enhanced with advanced features:
- Performance history tracking
- Smart optimization suggestions
- Alert thresholds
- Cost estimation
- CloudWatch query generation
"""

import json
import os
import time
import sys
from datetime import datetime
from logger import logger
from ucas_debugger import UCASDebugger, Colors
from input import extract_lambda_input
from data import get_akamai_onboarding_compliance
from validate import validate_akamai
from output import build_result, write_result

# Initialize Enhanced UCAS Debugger with auto-environment detection
is_local = os.getenv('AWS_EXECUTION_ENV') is None
ucas = UCASDebugger(logger, colored_output=is_local)


def handler(event, context):
    """
    Lambda handler for akamaivalidator with enhanced debugging.
    """
    request_id = context.request_id if context else "local"
    function_name = context.function_name if context else "local"
    ucas.method_start("handler", request_id=request_id, function_name=function_name)
    
    try:
        # Extract input parameters
        identifier = event.get('identifier')
        control = event.get('control')
        validator = event.get('validator', {})
        
        validator_type = validator.get('type')
        criteria = validator.get('criteria', [])
        
        ucas.log_metrics(
            identifier=identifier,
            control=control,
            validator_type=validator_type,
            criteria=criteria
        )
        
        # Step 1: Get Akamai compliance data
        result_data = get_akamai_onboarding_compliance(identifier)
        
        # Step 2: Validate against criteria
        validation_passed = validate_akamai(validator, result_data)
        
        # Step 3: Build result object
        result = build_result(identifier, control, validator, validation_passed)
        
        # Step 4: Write result to DynamoDB
        write_result(identifier, control, validator, result)
        
        # Log final status
        ucas.method_stop(
            "handler",
            final_status="PASS" if validation_passed else "FAIL",
            result_size=len(json.dumps(result))
        )
        
        # Finalize and save history
        ucas.finalize()
        
        return result
        
    except Exception as e:
        # Print clean error message
        ucas.print_error(e)
        
        # Log error to CloudWatch
        logger.error(f"Error in handler: {str(e)}", exc_info=True)
        
        # Stop method tracking with error info
        ucas.method_stop(
            "handler",
            final_status="ERROR",
            error_type=type(e).__name__,
            error_message=str(e)
        )
        
        # Handle differently based on environment
        if not is_local:
            raise  # Lambda runtime sees the error
        else:
            sys.exit(1)  # Clean exit locally
